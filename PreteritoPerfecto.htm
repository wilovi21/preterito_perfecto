                <h2 class="text-3xl font-semibold text-sky-700 mb-4">1. ¡Hola! Conectando el Pasado con el Presente</h2>

                <div class="space-y-4 text-lg leading-relaxed">

                    <p>¡Bienvenido/a a esta aventura por el Pretérito Perfecto! En esta sección inicial, descubrirás qué es exactamente este tiempo verbal y por qué es tan súper útil en nuestra comunicación diaria en español. Verás que es como un puente mágico que une lo que ya pasó con nuestro ahora.</p>

                    <p>El Pretérito Perfecto, a veces llamado Presente Perfecto, es un tiempo verbal fundamental en español. Aunque habla de acciones que ya sucedieron, su característica principal es que estas acciones tienen una <strong>conexión directa con el presente</strong>. No nos referimos a un momento específico en el pasado que ya terminó, sino a algo que ha ocurrido en un periodo de tiempo que aún no ha finalizado, o a una experiencia que sigue siendo relevante ahora.</p>

                    <p>Piensa en ello como una "noticia" del pasado que todavía está "fresca" o que tiene consecuencias en tu vida hoy. Por ejemplo, si dices "¡<strong>He comido</strong> un pastechi delicioso hoy!", esa acción de comer ya pasó, pero el "hoy" todavía continúa y quizás aún sientes el rico sabor. ¡Es por eso que lo usamos mucho para hablar de experiencias, logros, noticias o eventos recientes! ¿Listo/a para empezar?</p>

                </div>

                    <p>El Pretérito Perfecto se forma con dos partes, ¡como un equipo inseparable!</p>

                        <h3 class="text-2xl font-semibold text-amber-700 mb-3">Paso 2: Formación del Participio Pasado</h3>

                        <p class="mb-3">El participio pasado es la forma del verbo que generalmente termina en "-ado" o "-ido".</p>

                        <ul class="list-disc list-inside space-y-2 pl-4">

                            <li>Para verbos que terminan en <strong>-AR</strong> (como <em>hablar, estudiar, jugar</em>):

                                <ul class="list-circle list-inside pl-6">

                                    <li>Quita <strong>-AR</strong> y añade <strong>-ADO</strong>.</li>

                                    <li>Ejemplos: Hablar $\rightarrow$ habl<strong>ado</strong>; Estudiar $\rightarrow$ estudi<strong>ado</strong>; Jugar $\rightarrow$ jug<strong>ado</strong></li>

                                </ul>

                            </li>

                            <li>Para verbos que terminan en <strong>-ER</strong> (como <em>comer, aprender, leer</em>) o <strong>-IR</strong> (como <em>vivir, escribir, salir</em>):

                                <ul class="list-circle list-inside pl-6">

                                    <li>Quita <strong>-ER</strong> o <strong>-IR</strong> y añade <strong>-IDO</strong>.</li>

                                    <li>Ejemplos: Comer $\rightarrow$ com<strong>ido</strong>; Aprender $\rightarrow$ aprend<strong>ido</strong>; Vivir $\rightarrow$ viv<strong>ido</strong></li>

                                </ul>

                            </li>

                        </ul>

                    </div>

        async function callGeminiAPI(prompt) {

            llmLoading.style.display = 'block'; // Mostrar indicador de carga

            llmOutput.innerHTML = '<p class="text-slate-500">Generando respuesta...</p>'; // Mensaje de carga inicial


            let chatHistory = [];

            chatHistory.push({ role: "user", parts: [{ text: prompt }] });

            const payload = { contents: chatHistory };

            const apiKey = ""; // Si quieres usar modelos diferentes a gemini-2.0-flash o imagen-3.0-generate-002, proporciona una clave API aquí. De lo contrario, déjalo así.

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


            try {

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                const result = await response.json();


                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const text = result.candidates[0].content.parts[0].text;

                    llmOutput.innerHTML = `<p>${text}</p>`; // Mostrar la respuesta

                } else {

                    llmOutput.innerHTML = '<p class="text-red-500">Error: No se pudo generar la respuesta. Inténtalo de nuevo.</p>';

                }

            } catch (error) {

                console.error('Error al llamar a la API de Gemini:', error);

                llmOutput.innerHTML = '<p class="text-red-500">Error de conexión o de la API. Por favor, revisa tu conexión e inténtalo de nuevo.</p>';

            } finally {

                llmLoading.style.display = 'none'; // Ocultar indicador de carga

            }

        }

                const prompt = `Eres un tutor de español amigable para adolescentes. El usuario ha escrito la siguiente frase usando el Pretérito Perfecto: '${userInput}'. Asegúrate de mantener el Pretérito Perfecto donde sea apropiado.`;

                callGeminiAPI(prompt);

            } else {
